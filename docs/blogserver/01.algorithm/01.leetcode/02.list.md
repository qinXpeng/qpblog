## 1.反转链表
::: tip Describe
给你单链表的头节点 head ，请你反转链表，并返回反转后的链表。
::: right
[题目链接](https://leetcode.cn/problems/reverse-linked-list/)
:::

::: details Code
<code-group>
  <code-block title="C++" active>
  ```cpp
  /**
  * Definition for singly-linked list.
  * struct ListNode {
  *     int val;
  *     ListNode *next;
  *     ListNode() : val(0), next(nullptr) {}
  *     ListNode(int x) : val(x), next(nullptr) {}
  *     ListNode(int x, ListNode *next) : val(x), next(next) {}
  * };
  */
  class Solution {
  public:
      ListNode* reverseList(ListNode* head) {
          if(head == nullptr){
              return head;
          }
          ListNode *p = nullptr;
          while(head!=nullptr){
              ListNode *q=head;
              head=head->next;
              q->next=p;
              p=q;
          }
          return p;
      }
  };
  ```

## 3.k个一组反转链表
::: tip Describe
将链表k个节点一组反转，每次用尾巴连接新的一组反转链表，然后更新尾巴即可。
::: right
[题目链接](https://leetcode.cn/problems/reverse-nodes-in-k-group/)
:::

::: details Code
<code-group>
  <code-block title="C++" active>
  ```cpp
  /**
  * Definition for singly-linked list.
  * struct ListNode {
  *     int val;
  *     ListNode *next;
  *     ListNode() : val(0), next(nullptr) {}
  *     ListNode(int x) : val(x), next(nullptr) {}
  *     ListNode(int x, ListNode *next) : val(x), next(next) {}
  * };
  */
  class Solution {
  public:
      ListNode* felxList(ListNode* &head,int k){
          ListNode* p = NULL;
          ListNode* res = NULL;
          while(head!=NULL && k>0){
              p=head;
              head=head->next;
              p->next=res;
              res=p;
              k--;
          }
          return res;
      }
      int countList(ListNode* head){
          int len=0;
          while(head!=NULL){
              len++;
              head=head->next;
          }
          return len;
      }
      ListNode* reverseKGroup(ListNode* head, int k) {
          if(head==NULL){
              return head;
          }
          int n= countList(head);
          int rotates = n/k;
          ListNode* tail = new ListNode();
          ListNode* ans = tail;
          while(rotates--){
              ListNode* last=head; // 头节点最后是尾巴
              ListNode* resList = felxList(head,k); //反转链表
              tail->next = resList; // 用尾巴连接新的一组反转链表
              tail=last; // 更新尾巴
          }
          tail->next=head; // 更新尾巴
          return ans->next;
      }
  };
  ```

  ```go
      }
  ```

## 4.环形链表
::: tip Describe
判断链表中是否有环,并返回节点。
::: right
[题目链接](https://leetcode.cn/problems/linked-list-cycle-ii/)
:::

::: details Code
```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func detectCycle(head *ListNode) *ListNode {
    if head == nil {
        return head
    }
    low := head
    fast := head
    for ;fast != nil && fast.Next!=nil;{
        fast=fast.Next.Next
        low=low.Next
        if fast==low{
            for ;head!=low;{
                head=head.Next
                low=low.Next
            }
            return head
        }
    }
    return nil
}
```

::: tip Describe

::: right

::: details Code

```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func getIntersectionNode(headA, headB *ListNode) *ListNode {
    if headA==nil || headB ==nil {
        return nil
    }
    la := headA
    lb:=headB
    for ;la != lb ;{
        if la == nil{
            la = headB
        }else{
            la=la.Next
        }
        if lb == nil{
            lb = headA
        }else{
            lb=lb.Next
        }
    }
    return la
}
```

```go
func reverList(head **ListNode ,k int)*ListNode{
    var pa *ListNode
    var pb *ListNode
    for ;head!=nil&&k>0;{
        pb = *head
        *head=(*head).Next
        pb.Next=pa;
        pa=pb;
        k--
    }
    return pa

func reverseBetween(head *ListNode, left int, right int) *ListNode {
    if left==right {
        return head
    }
    res := head
    k := 0
    var last *ListNode
    for ;;{
        k++
        if k ==left{
            tail := head
            reverList := reverList(&head,right-left+1)
            tail.Next=head
            if left == 1 {
                return reverList
            }
            last.Next = reverList
            return res
        }
        last=head
        head=head.Next
    }
```

```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func mergeTwoList(la ,lb *ListNode)*ListNode{
    pa := new(ListNode)
    pb := pa
    for ;la!=nil&&lb!=nil;{
        if la.Val <= lb.Val{
            pb.Next=la
            la=la.Next
        }else{
            pb.Next=lb
            lb=lb.Next
        }
        pb=pb.Next
    }
    if la!=nil{
        pb.Next=la
    }
    if lb !=nil{
        pb.Next=lb
    }
    return pa.Next
}
func mergeLists(l,r int,lists []*ListNode)*ListNode{
    if l == r{
        return lists[l]
    }
    if l>r{
        return nil
    }
    mid := (l + r) >>1
    return mergeTwoList(mergeLists(l,mid,lists),mergeLists(mid+1,r,lists))
}
func mergeKLists(lists []*ListNode) *ListNode {
    n:=len(lists)
    if n == 0{
        return nil
    }
    return mergeLists(0,n-1,lists)
}
```

```go
func NewNode(val int) *Node{
	return &Node{
		Val: val,
		Next: nil,
		Random: nil,
	}
}
func copyRandomList(head *Node) *Node {
	if head == nil {
		return nil
	}
	p := head
	for ;head != nil ;{
		nd := NewNode(head.Val)
		nd.Next = head.Next
		head.Next = nd
		head=nd.Next
	}
	head = p
	for ;head!=nil;{
		// random
		nd := head.Next
		if head.Random != nil {
			hrnd := head.Random
			nd.Random = hrnd.Next
		}
        head=nd.Next
	}
    head = p
    q := p.Next
    for ;head!=nil;{
        // rotate
        nd := head.Next
		head.Next = nd.Next
		if nd.Next != nil {
			nd.Next = nd.Next.Next
		}
		head=head.Next
    }

	return q
}
```

```cpp
class Solution {
public:
	bool isPalindrome(ListNode *head) {
		if (head == NULL)
			return true;
		function<bool(ListNode *)> dfs = [&](ListNode *node) -> bool {
			if (node == NULL)
				return true;
			bool istrue = dfs(node->next);
			if (istrue) {
				istrue = head->val == node->val;
				head = head->next;
			}
			return istrue;
		};
		return dfs(head);
	}
};
```

## 10.LRU缓存

  ```cpp
class LRUCache {
   public:
    LRUCache(int capacity) { this->_capacity = capacity; }

    int get(int key) {
        if (this->_lru_map.find(key) != this->_lru_map.end()) {
            auto e = this->_lru_map[key];
            int val = e->second;
            this->_link.splice(this->_link.begin(), this->_link, e);  //将list _link 中e，插入到_link头节点前面
            return val;
        } else {
            return -1;
        }
    }
    void put(int key, int value) {
        if (this->_lru_map.find(key) != this->_lru_map.end()) {  // update key
            auto e = this->_lru_map[key];
            e->second = value;
            this->_link.splice(this->_link.begin(), this->_link, e);
        } else {
            if (this->_link.size() == this->_capacity) {  // pop last
                auto p = this->_link.back();
                int _key = p.first;
                this->_link.pop_back();
                this->_lru_map.erase(_key);
            }
            // insert front
            this->_link.push_front({key, value});
            this->_lru_map[key] = this->_link.begin();
        }
    }
   private:
    list<pair<int, int>> _link;
    unordered_map<int, list<pair<int, int>>::iterator> _lru_map;
    int _capacity;
};
  ```

  ```go
func init() { 
	debug.SetGCPercent(-1)
}
type LRUCache struct {
	MapCache map[int]*list.Element
	Link     *list.List
	Capacity int
}
type node struct {
	n_key   int
	n_value int
}
func Constructor(capacity int) LRUCache {
	return LRUCache{
		MapCache: make(map[int]*list.Element),
		Link:     list.New(),
		Capacity: capacity,
	}
}
func (this *LRUCache) getLen() int {
	return this.Link.Len()
}
func (this *LRUCache) Get(key int) int {
	e, ok := this.MapCache[key]
	if !ok {
		return -1
	}
	this.Link.MoveToFront(e)
	return e.Value.(node).n_value
}
func (this *LRUCache) Put(key int, value int) {
	e, ok := this.MapCache[key]
	if ok { // 存在key
		e.Value = node{key, value}
		this.Link.MoveToFront(e)
	} else {
		if this.getLen() == this.Capacity { //删除末端元素
			back := this.Link.Back()
			nkey := back.Value.(node).n_key
			delete(this.MapCache, nkey)
			this.Link.Remove(back)
		}
		this.Link.PushFront(node{
			n_key:   key,
			n_value: value,
		})
		inserEle := this.Link.Front()
		this.MapCache[key] = inserEle
	}
}
  ```
